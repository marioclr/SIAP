DROP PUBLIC DATABASE LINK SIAP_PRODUCCION;

--
-- SIAP_PRODUCCION  (Database Link) 
--
CREATE PUBLIC DATABASE LINK SIAP_PRODUCCION
 CONNECT TO SIAP
 IDENTIFIED BY <PWD>
 USING ' (DESCRIPTION=
    (ADDRESS=
      (PROTOCOL=TCP)
      (HOST=SIAPPRODUCCION)
      (PORT=1521)
    )
    (CONNECT_DATA=
      (SERVER=dedicated)
      (SERVICE_NAME=SIAP)
    )
  )';
